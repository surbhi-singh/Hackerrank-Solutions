import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int t = in.nextInt();
        for(int a0 = 0; a0 < t; a0++){
            String s = in.next();
            Stack stack = new Stack();
            char[] ch = s.toCharArray();
            int i =0;
            for(i =0;i<ch.length;i++)
                {
                if(ch[i] == '[' || ch[i] == '{' || ch[i] == '(')
                    stack.push(ch[i]);
                else if(ch[i] == ']')
                    {
                    if(!((char)stack.pop() == '['))
                    break;
                }
                else if(ch[i] == '}')
                    {
                    if(!((char)stack.pop() == '{'))
                    break;
                }
                else if(ch[i] == ')')
                    {
                    if(!((char)stack.pop() == '('))
                    break;
                }
            }
            if(i == ch.length)
                System.out.println("YES");
            else
                System.out.println("NO");
        }
    }
}
